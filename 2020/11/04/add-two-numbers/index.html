/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
var addTwoNumbers = function (l1, l2) {
  let result = temp = new ListNode()
  let exceed = 0

  while (l1 || l2 || exceed) {
    const sum = (l1 && l1.val || 0) + (l2 && l2.val || 0) + exceed
    const [e, n] = sum.toString().padStart(2, '0')

    l1 = l1 && l1.next
    l2 = l2 && l2.next

    exceed = Number(e)

    temp = temp.next = new ListNode(sum)
  }

  return result.next
}
